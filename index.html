Подключи Google Sheets — данные будут на всех устройствах.

1
Открой script.google.com → Новый проект → вставь код:
const SHEET_NAME = 'fitness'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(data.sheetId); const sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME); sheet.clearContents(); sheet.getRange(1,1).setValue(JSON.stringify(data.payload)); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const ss = SpreadsheetApp.openById(e.parameter.sheetId); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet||sheet.getLastRow()===0) return ContentService.createTextOutput(JSON.stringify({ok:true,data:null})).setMimeType(ContentService.MimeType.JSON); return ContentService.createTextOutput(JSON.stringify({ok:true,data:JSON.parse(sheet.getRange(1,1).getValue())})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON); } }
2
Развернуть → Новое развёртывание → Веб-приложение → "Кто имеет доступ": Все → Развернуть. Скопируй URL.
3
Открой sheets.new → скопируй ID из адресной строки (между /d/ и /edit).
